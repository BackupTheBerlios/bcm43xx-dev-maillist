 	u8 boardrev;
 	u8 locale:4;
 	u8 antennas_aphy:2;

 	value = sprom[BCM43xx_SPROM_ETHPHY];

-	bcm->sprom.et0mdcport = (value & (1 << 14)) >> 14;
